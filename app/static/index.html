class GlucosePredictor:
    def __init__(self):
        self.model = load_model('models/glucose_predictor.h5')
        self.scaler = StandardScaler()
    
    def predict(self, glucose_history, meals, activities):
        # Prepare sequence for LSTM
        sequence = self._prepare_features(glucose_history, meals)
        sequence = self.scaler.fit_transform(sequence)
        
        # Reshape for LSTM: (samples, timesteps, features)
        X = sequence.reshape(1, 24, 5)
        
        # Generate prediction
        prediction = self.model.predict(X)
        return self.scaler.inverse_transform(prediction)
